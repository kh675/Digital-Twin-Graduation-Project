HOW TO USE THE EMBEDDINGS
================================================================================

Example: Load and use student embeddings

import pickle
from sklearn.metrics.pairwise import cosine_similarity

# Load embeddings
with open('embeddings/embeddings_students.pkl', 'rb') as f:
    student_data = pickle.load(f)
    student_ids = student_data['ids']
    student_embeddings = student_data['embeddings']

with open('embeddings/embeddings_jobs.pkl', 'rb') as f:
    job_data = pickle.load(f)
    job_ids = job_data['ids']
    job_embeddings = job_data['embeddings']

# Find best job matches for a student
student_idx = 0  # First student
similarities = cosine_similarity(
    student_embeddings[student_idx:student_idx+1],
    job_embeddings
)[0]

# Get top 5 job matches
top_5_indices = similarities.argsort()[-5:][::-1]
for idx in top_5_indices:
    print(f"Job {job_ids[idx]}: {similarities[idx]:.4f}")
